#include <stdio.h>

int main()
{
    int Grade;
    printf("Enter Grade: ");
    scanf("%d", &Grade);
    
    if (Grade >=91)
    {
        printf ("Your letter grade is A");
    }
    else if (Grade >=84)
    {
        printf ("Your letter grade is A-");
    }
    else if (Grade >=76)
    {
        printf("Your letter grade is B");
    }
    else if (Grade >=68)
    {
        printf ("Your letter grade is B-");
    }
    else if (Grade >=60)
    {
        printf("Your letter grade is C");
    }
    else if (Grade >=50)
    {
        printf("Your letter grade is D");
    }
    else if (Grade < 50)
    {
        printf("Your letter grade is F");
    }
    else
    {
        printf("Invalid Input. Try again");
    }
        return 0;
}
